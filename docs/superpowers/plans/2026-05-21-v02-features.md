- [ ] **Step 2: Create EscalationEvaluator.swift**

```swift
public struct EscalationEvaluator: Sendable {
    public init() {}

    public func evaluate(_ findings: [Finding], rules: [EscalationRule]) -> [Finding] {
        var result = findings
        for rule in rules {
            result = apply(rule, to: result)
        }
        return result
    }

    private func apply(_ rule: EscalationRule, to findings: [Finding]) -> [Finding] {
        switch rule.scope {
        case .perServer:
            return applyPerServer(rule, to: findings)
        case .global:
            return applyGlobal(rule, to: findings)
        }
    }

    private func applyPerServer(_ rule: EscalationRule, to findings: [Finding]) -> [Finding] {
        // Group findings by (app, serverName)
        let groups = Dictionary(grouping: findings) { "\($0.app)||\($0.serverName ?? "")" }
        var result = findings
        for (_, group) in groups {
            let ruleIds = Set(group.map(\.ruleId))
            guard rule.requiresRuleIds.isSubset(of: ruleIds) else { continue }
            result = result.map { finding in
                guard
                    group.contains(finding),
                    rule.requiresRuleIds.contains(finding.ruleId),
                    finding.severity.sortRank > rule.escalateTo.sortRank
                else { return finding }
                return escalated(finding, to: rule)
            }
        }
        return result
    }

    private func applyGlobal(_ rule: EscalationRule, to findings: [Finding]) -> [Finding] {
        let allRuleIds = Set(findings.map(\.ruleId))
        guard rule.requiresRuleIds.isSubset(of: allRuleIds) else { return findings }

        if rule.minimumDistinctServers > 1 {
            // Each required ruleId must appear on at least minimumDistinctServers distinct servers
            for reqId in rule.requiresRuleIds {
                let distinctServers = Set(findings.filter { $0.ruleId == reqId }.compactMap(\.serverName))
                if distinctServers.count < rule.minimumDistinctServers { return findings }
            }
        }

        return findings.map { finding in
            guard
                rule.requiresRuleIds.contains(finding.ruleId),
                finding.severity.sortRank > rule.escalateTo.sortRank
            else { return finding }
            return escalated(finding, to: rule)
        }
    }

    private func escalated(_ finding: Finding, to rule: EscalationRule) -> Finding {
        Finding(
            ruleId: finding.ruleId,
            severity: rule.escalateTo,
            app: finding.app,
            serverName: finding.serverName,
            extensionId: finding.extensionId,
            affectedPath: finding.affectedPath,
            maskedValue: finding.maskedValue,
            title: finding.title,
            explanation: finding.explanation,
            recommendation: finding.recommendation,
            escalationReason: rule.reason
        )
    }
}
```

- [ ] **Step 2: Create EscalationEvaluator.cs**

```csharp
using AIExposureScanner.Scanner.Models;

namespace AIExposureScanner.Scanner;

public sealed class EscalationEvaluator
{
    public IReadOnlyList<Finding> Evaluate(IReadOnlyList<Finding> findings, IReadOnlyList<EscalationRule> rules)
    {
        var result = findings.ToList();
        foreach (var rule in rules)
        {
            result = Apply(rule, result);
        }
        return result;
    }

    private static List<Finding> Apply(EscalationRule rule, List<Finding> findings) =>
        rule.Scope switch
        {
            EscalationScope.PerServer => ApplyPerServer(rule, findings),
            EscalationScope.Global => ApplyGlobal(rule, findings),
            _ => findings
        };

    private static List<Finding> ApplyPerServer(EscalationRule rule, List<Finding> findings)
    {
        var groups = findings
            .GroupBy(f => $"{f.App}||{f.ServerName ?? string.Empty}", StringComparer.Ordinal)
            .ToList();

        var result = findings.ToList();
        foreach (var group in groups)
        {
            var ruleIds = group.Select(f => f.RuleId).ToHashSet(StringComparer.Ordinal);
            if (!rule.RequiresRuleIds.IsSubsetOf(ruleIds)) continue;

            result = result
                .Select(f =>
                    group.Contains(f) &&
                    rule.RequiresRuleIds.Contains(f.RuleId) &&
                    (int)f.Severity > (int)rule.EscalateTo
                        ? f with { Severity = rule.EscalateTo, EscalationReason = rule.Reason }
                        : f)
                .ToList();
        }
        return result;
    }

    private static List<Finding> ApplyGlobal(EscalationRule rule, List<Finding> findings)
    {
        var allRuleIds = findings.Select(f => f.RuleId).ToHashSet(StringComparer.Ordinal);
        if (!rule.RequiresRuleIds.IsSubsetOf(allRuleIds)) return findings;

        if (rule.MinimumDistinctServers > 1)
        {
            foreach (var reqId in rule.RequiresRuleIds)
            {
                var distinctServers = findings
                    .Where(f => f.RuleId == reqId && f.ServerName is not null)
                    .Select(f => f.ServerName!)
                    .Distinct(StringComparer.Ordinal)
                    .Count();
                if (distinctServers < rule.MinimumDistinctServers) return findings;
            }
        }

        return findings
            .Select(f =>
                rule.RequiresRuleIds.Contains(f.RuleId) && (int)f.Severity > (int)rule.EscalateTo
                    ? f with { Severity = rule.EscalateTo, EscalationReason = rule.Reason }
                    : f)
            .ToList();
    }
}
```

```csharp
public async Task<ScanResult> ScanAsync(IFilesystemFacade fs, CancellationToken ct = default, IReadOnlyList<RulePack>? packs = null)
{
    packs ??= [];
    var tasks = Detectors.Select(async detector =>
    {
        using var cts = CancellationTokenSource.CreateLinkedTokenSource(ct);
        cts.CancelAfter(TimeSpan.FromSeconds(5));
        try { return await Task.Run(() => detector.CollectFacts(fs), cts.Token); }
        catch (OperationCanceledException) { return new ScanFacts(); }
        catch (Exception) { return new ScanFacts(); }
    });

    var allFacts = await Task.WhenAll(tasks);
    var merged = new ScanFacts();
    foreach (var f in allFacts) merged.Append(f);

    var builtInFindings = Evaluator.Evaluate(merged);
    var allFindings = new RulePackEvaluator().Evaluate(merged, packs, builtInFindings);
```

```swift
public func json(scanResult: ScanResult, platform: String = "macos", scannedAt: Date = Date()) -> String {
    let summary = ReportSummary(scanResult: scanResult)

    // Lightweight Encodable DTO — no Codable changes needed on ScanResult
    struct JsonSummary: Encodable {
        let toolsFound: Int
        let mcpServersFound: Int
        let critical: Int
        let high: Int

    }

        let affectedPath: String?
        let maskedValue: String?
        let title: String

        let recommendation: String
        let escalationReason: String?
    }
    struct JsonReport: Encodable {
        let schemaVersion = "1.0.0"
        let scannerVersion = "0.2.0"
        let scannedAt: String
        let platform: String
        let status: String
        let summary: JsonSummary
        let findings: [JsonFinding]
    }

    let report = JsonReport(
        scannedAt: Self.iso8601(scannedAt),
        platform: platform,
        status: summary.status.rawValue,
        summary: JsonSummary(
            toolsFound: summary.toolsFound,
            mcpServersFound: summary.mcpServersFound,
            critical: summary.critical,
            high: summary.high,
            medium: summary.medium,
            low: summary.low
        ),
        findings: scanResult.findings.map { f in
            JsonFinding(
                ruleId: f.ruleId,
                severity: f.severity.rawValue,
                app: f.app,
                serverName: f.serverName,
                extensionId: f.extensionId,
                affectedPath: f.affectedPath,
                maskedValue: f.maskedValue,
                title: f.title,
                explanation: f.explanation,
                recommendation: f.recommendation,
                escalationReason: f.escalationReason
            )
        }
    )

    let encoder = JSONEncoder()
    encoder.outputFormatting = [.prettyPrinted, .sortedKeys, .withoutEscapingSlashes]
```

```swift
    /// Parse and validate a YAML string. Returns .valid or .invalid.
    public static func load(yaml yamlString: String) -> RulePackLoadResult {
        do {
            let raw = try YAMLDecoder().decode(RawPack.self, from: yamlString)
            return try validate(raw)
        } catch let error as DecodingError {
            return .invalid("YAML parse error: \(error.localizedDescription)")
        } catch let error as RulePackError {
            return .invalid(error.message)
        } catch {
            return .invalid(error.localizedDescription)
        }
    }

        let message: String
    }

    private static func validate(_ raw: RawPack) throws -> RulePackLoadResult {
        if raw.id.hasPrefix("AES-") {
            return .invalid("Pack id '\(raw.id)' must not start with 'AES-' (reserved namespace)")
        }
        var rules: [PackRule] = []
        for r in raw.rules ?? [] {
            if r.id.hasPrefix("AES-") {
                return .invalid("Rule id '\(r.id)' must not start with 'AES-' (reserved namespace)")
            }
            guard let severity = Severity(rawValue: r.severity.lowercased()) else {
                return .invalid("Rule '\(r.id)' has invalid severity '\(r.severity)'")
            }

                return .invalid("Rule '\(r.id)' has unknown fact type '\(r.match.fact)'")
            }
            let match = buildMatch(raw: r.match, factType: factType)
            guard match.hasAtLeastOneCondition else {
                return .invalid("Rule '\(r.id)' match must have at least one condition")
            }
            rules.append(PackRule(id: r.id, severity: severity, title: r.title,
                                  explanation: r.explanation, recommendation: r.recommendation, match: match))
        }
        let overrides = (raw.overrides ?? []).compactMap { o -> PackOverride? in
            guard let sev = Severity(rawValue: o.severity.lowercased()) else { return nil }
            return PackOverride(id: o.id, severity: sev)

        let escalations = (raw.escalations ?? []).map { e in

        }

            key: raw.key,
```

```csharp
public static class RulePackLoader
{
    private static readonly IDeserializer Deserializer = new DeserializerBuilder()
        .WithNamingConvention(UnderscoredNamingConvention.Instance)
        .IgnoreUnmatchedProperties()
        .Build();

    public static RulePackLoadResult Load(string yaml)
    {
        try
        {
            var raw = Deserializer.Deserialize<RawPack>(yaml);
            return Validate(raw);
        }
        catch (Exception ex)
        {
            return RulePackLoadResult.Invalid($"YAML parse error: {ex.Message}");
        }
    }

    private static RulePackLoadResult Validate(RawPack raw)
    {
        if (raw.Id.StartsWith("AES-", StringComparison.Ordinal))
            return RulePackLoadResult.Invalid($"Pack id '{raw.Id}' must not start with 'AES-'");

        var rules = new List<PackRule>();
        foreach (var r in raw.Rules ?? [])
        {
            if (r.Id.StartsWith("AES-", StringComparison.Ordinal))
                return RulePackLoadResult.Invalid($"Rule id '{r.Id}' must not start with 'AES-'");

            if (!Enum.TryParse<Severity>(r.Severity, ignoreCase: true, out var severity))
                return RulePackLoadResult.Invalid($"Rule '{r.Id}' has invalid severity '{r.Severity}'");

            var match = r.Match;
            if (!HasAnyCondition(match))
                return RulePackLoadResult.Invalid($"Rule '{r.Id}' match must have at least one condition");

            rules.Add(new PackRule(r.Id, r.Severity, r.Title, r.Explanation, r.Recommendation,
                new PackRuleMatch(
                    match.Fact,
                    match.CommandEquals, match.CommandContains, match.NameContains,
                    match.ArgsContain, match.App, match.Key, match.ValueEquals,
                    match.ExtensionIdContains
                )));
        }

        var overrides = (raw.Overrides ?? [])
            .Select(o => new PackOverride(o.Id, o.Severity))
            .ToList();

        var escalations = (raw.Escalations ?? [])
            .Select(e => new PackEscalation(e.Requires ?? [], e.Scope, e.EscalateTo, e.Reason))
            .ToList();

        return RulePackLoadResult.Valid(new RulePack(raw.Version, raw.Id, raw.Name, rules, overrides, escalations));
    }
```

```swift
public struct RulePackEvaluator: Sendable {
    public init() {}

    public func evaluate(
        facts: ScanFacts,
        packs: [RulePack],
        existingFindings: [Finding]
    ) -> [Finding] {
        // 1. Apply severity overrides to built-in findings
        var result = applyOverrides(existingFindings, packs: packs)
        // 2. Add findings from pack rules
        for pack in packs {
            result += packFindings(from: facts, pack: pack)
        }
        return result
    }

    // MARK: - Overrides

    private func applyOverrides(_ findings: [Finding], packs: [RulePack]) -> [Finding] {
        var overrideMap: [String: Severity] = [:]
        for pack in packs {
            for override in pack.overrides {
                overrideMap[override.id] = override.severity
            }
        }
        return findings.map { finding in
            guard let newSev = overrideMap[finding.ruleId],
                  finding.escalationReason == nil else { return finding }
            return Finding(ruleId: finding.ruleId, severity: newSev, app: finding.app,
                           serverName: finding.serverName, extensionId: finding.extensionId,
                           affectedPath: finding.affectedPath, maskedValue: finding.maskedValue,
                           title: finding.title, explanation: finding.explanation,
                           recommendation: finding.recommendation)
        }
    }

    // MARK: - Pack rule matching

    private func packFindings(from facts: ScanFacts, pack: RulePack) -> [Finding] {
        pack.rules.flatMap { matchFindings(from: facts, rule: $0) }
    }

    private func matchFindings(from facts: ScanFacts, rule: PackRule) -> [Finding] {
        switch rule.match.fact {
        case .mcpServer:
            return facts.mcpServers.compactMap { server in
                guard matchesMcpServer(server, match: rule.match) else { return nil }
                return Finding(ruleId: rule.id, severity: rule.severity, app: server.appId,
                               serverName: server.name, affectedPath: server.configPath,
                               title: rule.title, explanation: rule.explanation,
                               recommendation: rule.recommendation)
            }
        case .setting:
            return facts.settings.compactMap { setting in
                guard matchesSetting(setting, match: rule.match) else { return nil }
                return Finding(ruleId: rule.id, severity: rule.severity, app: setting.appId,
                               affectedPath: setting.configPath,
                               title: rule.title, explanation: rule.explanation,
                               recommendation: rule.recommendation)
            }
        case .extension:
            return facts.extensions.compactMap { ext in
                guard matchesExtension(ext, match: rule.match) else { return nil }
                return Finding(ruleId: rule.id, severity: rule.severity, app: ext.appId,
                               extensionId: ext.extensionId,
                               title: rule.title, explanation: rule.explanation,

        case .authFile:

                guard matchesAuthFile(auth, match: rule.match) else { return nil }
                return Finding(ruleId: rule.id, severity: rule.severity, app: auth.appId,
                               title: rule.title, explanation: rule.explanation,
                               recommendation: rule.recommendation)
```

```csharp
using AIExposureScanner.Scanner.Models;

namespace AIExposureScanner.Scanner.RulePack;

public sealed class RulePackEvaluator
{
    public IReadOnlyList<Finding> Evaluate(
        ScanFacts facts,
        IReadOnlyList<RulePack> packs,
        IReadOnlyList<Finding> existingFindings)
    {
        var result = ApplyOverrides(existingFindings, packs).ToList();
        foreach (var pack in packs)
        {
            result.AddRange(PackFindings(facts, pack));
        }
        return result;
    }

    private static IEnumerable<Finding> ApplyOverrides(IReadOnlyList<Finding> findings, IReadOnlyList<RulePack> packs)
    {
        var overrideMap = packs
            .SelectMany(p => p.Overrides)
            .ToDictionary(o => o.Id, o => o.Severity, StringComparer.Ordinal);

        return findings.Select(f =>
            overrideMap.TryGetValue(f.RuleId, out var newSev) && f.EscalationReason is null
                ? f with { Severity = Enum.Parse<Severity>(newSev, ignoreCase: true) }
                : f);
    }

    private static IEnumerable<Finding> PackFindings(ScanFacts facts, RulePack pack)
    {
        return pack.Rules.SelectMany(rule => MatchFindings(facts, rule));
    }

    private static IEnumerable<Finding> MatchFindings(ScanFacts facts, PackRule rule)
    {
        var m = rule.Match;
        return m.Fact.ToLowerInvariant() switch
        {
            "mcp_server" => facts.McpServers
                .Where(s => MatchesMcp(s, m))
                .Select(s => new Finding(rule.Id, ParseSev(rule.Severity), s.AppId,
                    ServerName: s.Name, AffectedPath: s.ConfigPath,
                    Title: rule.Title, Explanation: rule.Explanation,
                    Recommendation: rule.Recommendation)),
            "setting" => facts.Settings
                .Where(s => MatchesSetting(s, m))
                .Select(s => new Finding(rule.Id, ParseSev(rule.Severity), s.AppId,
                    AffectedPath: s.ConfigPath,
                    Title: rule.Title, Explanation: rule.Explanation,
                    Recommendation: rule.Recommendation)),
            "extension" => facts.Extensions
                .Where(e => MatchesExtension(e, m))
                .Select(e => new Finding(rule.Id, ParseSev(rule.Severity), e.AppId,
                    ExtensionId: e.ExtensionId,
                    Title: rule.Title, Explanation: rule.Explanation,
                    Recommendation: rule.Recommendation)),
            "auth_file" => facts.AuthFiles
                .Where(a => MatchesAuthFile(a, m))
                .Select(a => new Finding(rule.Id, ParseSev(rule.Severity), a.AppId,
                    Title: rule.Title, Explanation: rule.Explanation,
                    Recommendation: rule.Recommendation)),
            _ => []
        };

    private static bool MatchesMcp(McpServerFact s, PackRuleMatch m)

        if (m.App is not null && !s.AppId.Equals(m.App, StringComparison.OrdinalIgnoreCase)) return false;
        if (m.CommandEquals is not null && !s.Command.Equals(m.CommandEquals, StringComparison.OrdinalIgnoreCase)) return false;
        if (m.CommandContains is not null && !s.Command.Contains(m.CommandContains, StringComparison.OrdinalIgnoreCase)) return false;
        if (m.NameContains is not null && !s.Name.Contains(m.NameContains, StringComparison.OrdinalIgnoreCase)) return false;
        if (m.ArgsContain is not null)

            if (!joined.Contains(m.ArgsContain, StringComparison.OrdinalIgnoreCase)) return false;
        }
        return true;
    }

    private static bool MatchesSetting(SettingFact s, PackRuleMatch m)

        if (m.Key is not null && !s.Key.Equals(m.Key, StringComparison.OrdinalIgnoreCase)) return false;
        if (m.ValueEquals is not null)

            if (s.StringValue is null) return false;
            if (!s.StringValue.Equals(m.ValueEquals, StringComparison.OrdinalIgnoreCase)) return false;
        }
        return true;
    }
```

```swift
final class RulePackStore: ObservableObject {
    static let shared = RulePackStore()
    private static let defaultsKey = "loadedRulePacks"

    @Published private(set) var entries: [(yaml: String, result: RulePackLoadResult)] = []

    init() { reload() }

    func reload() {
        let yamls = UserDefaults.standard.stringArray(forKey: Self.defaultsKey) ?? []
        entries = yamls.map { ($0, RulePackLoader.load(yaml: $0)) }
    }

    func add(yaml: String) {
        var yamls = UserDefaults.standard.stringArray(forKey: Self.defaultsKey) ?? []
        yamls.append(yaml)
        UserDefaults.standard.set(yamls, forKey: Self.defaultsKey)
        reload()
    }

    func remove(at index: Int) {
        var yamls = UserDefaults.standard.stringArray(forKey: Self.defaultsKey) ?? []
        guard index < yamls.count else { return }
        yamls.remove(at: index)
        UserDefaults.standard.set(yamls, forKey: Self.defaultsKey)
        reload()
    }

    var validPacks: [RulePack] {
        entries.compactMap {
            if case .valid(let p) = $0.result { return p }

        }
    }
```

```csharp
internal sealed class RulePackStore
{
    private static readonly string StorePath = Path.Combine(
        Environment.GetFolderPath(Environment.SpecialFolder.LocalApplicationData),
        "AIExposureScanner", "rule-packs.json");

    public List<(string Yaml, RulePackLoadResult Result)> Entries { get; private set; } = [];

    public void Load()
    {
        var yamls = new List<string>();
        if (File.Exists(StorePath))
        {
            try
            {
                yamls = JsonSerializer.Deserialize<List<string>>(File.ReadAllText(StorePath)) ?? [];
            }
            catch { /* corrupt file — start fresh */ }
        }
        Entries = yamls.Select(y => (y, RulePackLoader.Load(y))).ToList();
    }

    public void Add(string yaml)
    {
        var yamls = Entries.Select(e => e.Yaml).ToList();
        yamls.Add(yaml);
        Save(yamls);
        Load();
    }

    public void Remove(int index)
    {
        var yamls = Entries.Select(e => e.Yaml).ToList();
        if (index < yamls.Count) yamls.RemoveAt(index);
        Save(yamls);
        Load();
    }

    public IReadOnlyList<RulePack> ValidPacks =>
        Entries
            .Where(e => e.Result is RulePackLoadResult.Valid)
            .Select(e => ((RulePackLoadResult.Valid)e.Result).Pack)
            .ToArray();

    private static void Save(List<string> yamls)
    {
        Directory.CreateDirectory(Path.GetDirectoryName(StorePath)!);
        File.WriteAllText(StorePath, JsonSerializer.Serialize(yamls));
    }
}
```

```csharp
private void ShowSettings()
{
    var dialog = new Window
    {
        Title = "Settings — Rule Packs",
        Width = 520,
        Height = 420,
        Owner = this,
        WindowStartupLocation = WindowStartupLocation.CenterOwner
    };

    var stack = new StackPanel { Margin = new Thickness(16) };

    void Refresh()
    {
        stack.Children.Clear();
        stack.Children.Add(new TextBlock { Text = "Rule Packs", FontSize = 16, FontWeight = FontWeights.SemiBold, Margin = new Thickness(0, 0, 0, 12) });

        for (var i = 0; i < _packStore.Entries.Count; i++)
        {
            var (_, result) = _packStore.Entries[i];
            var capturedIndex = i;
            var row = new DockPanel { Margin = new Thickness(0, 0, 0, 8) };
            var removeBtn = new Button { Content = "Remove", Padding = new Thickness(8, 2, 8, 2) };
            removeBtn.Click += (_, _) => { _packStore.Remove(capturedIndex); Refresh(); };
            DockPanel.SetDock(removeBtn, Dock.Right);
            row.Children.Add(removeBtn);

            if (result is RulePackLoadResult.Valid v)
            {
                var info = new StackPanel();
                info.Children.Add(new TextBlock { Text = v.Pack.Name, FontWeight = FontWeights.Medium });
                info.Children.Add(new TextBlock { Text = $"{v.Pack.Rules.Count} rules · {v.Pack.Overrides.Count} overrides", FontSize = 11, Foreground = Brushes.Gray });
                row.Children.Add(info);
            }
            else if (result is RulePackLoadResult.Invalid inv)
            {
                row.Children.Add(new TextBlock { Text = $"⚠ Error: {inv.Error}", Foreground = Brushes.OrangeRed, TextWrapping = TextWrapping.Wrap });
            }

            stack.Children.Add(row);
        }

        var addBtn = new Button { Content = "Add Rule Pack…", Margin = new Thickness(0, 8, 0, 0) };
        addBtn.Click += (_, _) =>
        {
            var picker = new OpenFileDialog { Filter = "YAML files (*.yaml;*.yml)|*.yaml;*.yml", Title = "Select Rule Pack" };
            if (picker.ShowDialog() == true)
            {
                var yaml = File.ReadAllText(picker.FileName);
                _packStore.Add(yaml);
                Refresh();
            }
        };
        stack.Children.Add(addBtn);
    }

    Refresh();
    dialog.Content = new ScrollViewer { Content = stack };
    dialog.ShowDialog();
}
```
